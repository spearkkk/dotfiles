# Dot Utils Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Lua-based `u` utility CLI for timestamps, case conversion, and random values, exposed through `.local/bin/u` for fish and Alfred.

**Architecture:** `.local/bin/u` is a thin Bash launcher that resolves the stowed runtime path and execs Lua. `.local/share/dot-utils/u.lua` dispatches subcommands into focused modules under `.local/share/dot-utils/lib/`. A Bash test script verifies CLI behavior end to end with no extra test framework.

**Tech Stack:** Lua 5.4, Bash, macOS `/dev/urandom`, macOS `/usr/bin/perl` with a safe seconds-only fallback for millisecond timestamps.

---

## File Structure

- Create: `.local/bin/u`
  - Stable executable entrypoint for fish, shell scripts, and Alfred.
  - Locates Lua with `DOT_UTILS_LUA` override or `command -v lua`.
  - Runs `.local/share/dot-utils/u.lua` from either `$HOME` after stow or the repo checkout.

- Create: `.local/share/dot-utils/u.lua`
  - Top-level CLI dispatcher.
  - Owns global usage text, group lookup, exit codes, and exception handling.
  - Loads modules from `.local/share/dot-utils/lib/`.

- Create: `.local/share/dot-utils/lib/timestamp.lua`
  - Implements `u ts now`, `u ts ms`, `u ts iso`, and `u ts date`.

- Create: `.local/share/dot-utils/lib/case.lua`
  - Implements `u case snake`, `kebab`, `camel`, `pascal`, `upper`, and `lower`.
  - Treats all remaining CLI arguments as one space-joined input string.

- Create: `.local/share/dot-utils/lib/random.lua`
  - Implements `u rand hex [bytes]`, `u rand id [length]`, and `u rand var [style]`.
  - Reads random bytes from `/dev/urandom`.

- Create: `scripts/test-dot-utils.sh`
  - Lightweight Bash test runner for the CLI.
  - Uses regex and equality assertions.
  - Exits non-zero on failures.

- Modify: `README.md`
  - Adds concise usage examples for `u`.

```bash
#!/usr/bin/env bash

set -u

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
U="$ROOT_DIR/.local/bin/u"
FAILURES=0

fail() {
  printf 'not ok - %s\n' "$1" >&2
  FAILURES=$((FAILURES + 1))
}

pass() {
  printf 'ok - %s\n' "$1"
}

assert_eq() {
  local name="$1"
  local expected="$2"
  local actual="$3"

  if [ "$actual" = "$expected" ]; then
    pass "$name"
  else
    fail "$name: expected '$expected', got '$actual'"
  fi
}

assert_contains() {
  local name="$1"
  local actual="$2"
  local expected_substring="$3"

  case "$actual" in
    *"$expected_substring"*) pass "$name" ;;
    *) fail "$name: expected output to contain '$expected_substring', got '$actual'" ;;
  esac
}

assert_match() {
  local name="$1"
  local actual="$2"
  local regex="$3"

  if [[ "$actual" =~ $regex ]]; then
    pass "$name"
  else
    fail "$name: expected '$actual' to match /$regex/"
  fi
}

assert_status() {
  local name="$1"
  local expected="$2"
  local actual="$3"

  if [ "$actual" -eq "$expected" ]; then
    pass "$name"
  else
    fail "$name: expected status $expected, got $actual"
  fi
}

require_executable() {
  if [ -x "$U" ]; then
    pass "u launcher is executable"
    return 0
  fi

  fail "$U is not executable"
  return 1
}

main() {
  local tmp_out
  local tmp_err
  tmp_out="$(mktemp)"
  tmp_err="$(mktemp)"
  trap 'rm -f "$tmp_out" "$tmp_err"' EXIT

  require_executable || exit 1

  local help_output
  help_output="$("$U" help)"
  assert_contains "help shows usage" "$help_output" "Usage: u <group> <command> [args...]"

  "$U" nope >"$tmp_out" 2>"$tmp_err"
  local status=$?
  assert_status "unknown group exits 2" 2 "$status"
  assert_contains "unknown group reports group name" "$(cat "$tmp_err")" "u: unknown group 'nope'"

  if [ "$FAILURES" -gt 0 ]; then
    printf '%s failure(s)\n' "$FAILURES" >&2
    exit 1
  fi
}

main "$@"
```

```lua
local function script_dir()
  local source = arg and arg[0] or "."
  return source:match("^(.*)/[^/]*$") or "."
end

local root = script_dir()
package.path = root .. "/?.lua;" .. root .. "/?/init.lua;" .. root .. "/lib/?.lua;" .. package.path

local USAGE = [[Usage: u <group> <command> [args...]

Groups:
  ts     Timestamp helpers
  case   Case conversion helpers
  rand   Random value helpers

Examples:
  u ts now
  u case camel "hello world"
  u rand hex 8
]]

local groups = {}

local function sub_args(argv, start_index)
  local out = {}
  for i = start_index, #argv do
    out[#out + 1] = argv[i]
  end
  return out
end

local function write_usage(stream)
  stream:write(USAGE)
end

local function main(argv)
  local group_name = argv[1]

  if group_name == nil or group_name == "help" or group_name == "--help" or group_name == "-h" then
    write_usage(io.stdout)
    return 0
  end

  local group = groups[group_name]
  if group == nil then
    io.stderr:write("u: unknown group '" .. group_name .. "'\n")
    write_usage(io.stderr)
    return 2
  end

  local ok, success, output, exit_code = pcall(group.run, argv[2], sub_args(argv, 3))
  if not ok then
    io.stderr:write("u: " .. tostring(success) .. "\n")
    return 1
  end

  if not success then
    io.stderr:write(output .. "\n")
    return exit_code or 1
  end

  if output ~= nil and output ~= "" then
    io.stdout:write(output .. "\n")
  end

  return exit_code or 0
end

os.exit(main(arg or {}))
```

```lua
local M = {}

local USAGE = [[Usage: u ts <command>

Commands:
  now   Unix timestamp in seconds
  ms    Unix timestamp in milliseconds
  iso   Local ISO-like timestamp
  date  Local date as YYYY-MM-DD
]]

local function trim(value)
  return (value:gsub("%s+$", ""))
end

local function read_command(command)
  local handle = io.popen(command)
  if handle == nil then
    return nil
  end

  local output = handle:read("*a")
  local ok = handle:close()
  if not ok then
    return nil
  end

  return trim(output)
end

local function millis()
  local value = read_command("/usr/bin/perl -MTime::HiRes=time -e 'printf \"%.0f\\n\", time() * 1000'")
  if value ~= nil and value:match("^%d+$") then
    return value
  end

  return string.format("%.0f", os.time() * 1000)
end

function M.run(command, _args)
  if command == "help" or command == "--help" or command == "-h" then
    return true, USAGE, 0
  end

  if command == "now" then
    return true, tostring(os.time()), 0
  end

  if command == "ms" then
    return true, millis(), 0
  end

  if command == "iso" then
    return true, os.date("%Y-%m-%dT%H:%M:%S%z"), 0
  end

  if command == "date" then
    return true, os.date("%Y-%m-%d"), 0
  end

  if command == nil then
    return false, "u ts: missing command\n" .. USAGE, 2
  end

  return false, "u ts: unknown command '" .. command .. "'\n" .. USAGE, 2
end

return M
```

```lua
local M = {}

local USAGE = [[Usage: u case <command> <text...>

Commands:
  snake   Convert text to snake_case
  kebab   Convert text to kebab-case
  camel   Convert text to camelCase
  pascal  Convert text to PascalCase
  upper   Convert text to uppercase
  lower   Convert text to lowercase
]]

local function join_args(args)
  return table.concat(args, " ")
end

local function capitalize(word)
  if word == "" then
    return word
  end

  return word:sub(1, 1):upper() .. word:sub(2)
end

local function words(input)
  local normalized = input
    :gsub("([a-z0-9])([A-Z])", "%1 %2")
    :gsub("[%W_]+", " ")

  local out = {}
  for word in normalized:gmatch("%w+") do
    out[#out + 1] = word:lower()
  end

  return out
end

function M.to_snake(input)
  return table.concat(words(input), "_")
end

function M.to_kebab(input)
  return table.concat(words(input), "-")
end

function M.to_camel(input)
  local parts = words(input)
  if #parts == 0 then
    return ""
  end

  local out = { parts[1] }
  for i = 2, #parts do
    out[#out + 1] = capitalize(parts[i])
  end

  return table.concat(out, "")
end

function M.to_pascal(input)
  local out = {}
  for _, word in ipairs(words(input)) do
    out[#out + 1] = capitalize(word)
  end

  return table.concat(out, "")
end

function M.run(command, args)
  if command == "help" or command == "--help" or command == "-h" then
    return true, USAGE, 0
  end

  if command == nil then
    return false, "u case: missing command\n" .. USAGE, 2
  end

  local input = join_args(args)

  if command == "snake" then
    return true, M.to_snake(input), 0
  end

  if command == "kebab" then
    return true, M.to_kebab(input), 0
  end

  if command == "camel" then
    return true, M.to_camel(input), 0
  end

  if command == "pascal" then
    return true, M.to_pascal(input), 0
  end

  if command == "upper" then
    return true, input:upper(), 0
  end

  if command == "lower" then
    return true, input:lower(), 0
  end

  return false, "u case: unknown command '" .. command .. "'\n" .. USAGE, 2
end

return M
```

```lua
local M = {}

local USAGE = [[Usage: u rand <command> [args]

Commands:
  hex [bytes]     Lowercase hex string, default 8 bytes
  id [length]     Alphanumeric id, default 12 chars
  var [style]     Variable-like name; styles: camel, snake, kebab
]]

local ALPHANUM = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
local ADJECTIVES = { "bright", "calm", "clear", "fresh", "kind", "quick", "sharp", "silent", "steady", "tidy" }
local NOUNS = { "anchor", "branch", "field", "harbor", "lantern", "matrix", "signal", "thread", "vector", "window" }

local function random_bytes(count)
  local handle = io.open("/dev/urandom", "rb")
  if handle == nil then
    return nil, "u rand: cannot open /dev/urandom"
  end

  local bytes = handle:read(count)
  handle:close()

  if bytes == nil or #bytes ~= count then
    return nil, "u rand: could not read enough random bytes"
  end

  return bytes, nil
end

local function parse_count(raw, default, label, max)
  if raw == nil or raw == "" then
    return default, nil
  end

  local value = tonumber(raw)
  if value == nil or value % 1 ~= 0 or value < 1 or value > max then
    return nil, "u rand " .. label .. ": " .. (label == "hex" and "bytes" or "length") .. " must be between 1 and " .. tostring(max)
  end

  return value, nil
end

local function to_hex(bytes)
  local out = {}
  for i = 1, #bytes do
    out[#out + 1] = string.format("%02x", bytes:byte(i))
  end

  return table.concat(out, "")
end

local function from_alphabet(length, alphabet)
  local bytes, err = random_bytes(length)
  if bytes == nil then
    return nil, err
  end

  local out = {}
  for i = 1, #bytes do
    local index = (bytes:byte(i) % #alphabet) + 1
    out[#out + 1] = alphabet:sub(index, index)
  end

  return table.concat(out, ""), nil
end

local function pick(words, byte)
  local index = (byte % #words) + 1
  return words[index]
end

local function capitalize(word)
  return word:sub(1, 1):upper() .. word:sub(2)
end

local function random_var(style)
  local bytes, err = random_bytes(2)
  if bytes == nil then
    return nil, err
  end

  local first = pick(ADJECTIVES, bytes:byte(1))
  local second = pick(NOUNS, bytes:byte(2))

  if style == nil or style == "" or style == "camel" then
    return first .. capitalize(second), nil
  end

  if style == "snake" then
    return first .. "_" .. second, nil
  end

  if style == "kebab" then
    return first .. "-" .. second, nil
  end

  return nil, "u rand var: style must be one of camel, snake, kebab"
end

function M.run(command, args)
  if command == "help" or command == "--help" or command == "-h" then
    return true, USAGE, 0
  end

  if command == nil then
    return false, "u rand: missing command\n" .. USAGE, 2
  end

  if command == "hex" then
    local count, count_err = parse_count(args[1], 8, "hex", 1024)
    if count == nil then
      return false, count_err, 2
    end

    local bytes, err = random_bytes(count)
    if bytes == nil then
      return false, err, 1
    end

    return true, to_hex(bytes), 0
  end

  if command == "id" then
    local length, count_err = parse_count(args[1], 12, "id", 256)
    if length == nil then
      return false, count_err, 2
    end

    local value, err = from_alphabet(length, ALPHANUM)
    if value == nil then
      return false, err, 1
    end

    return true, value, 0
  end

  if command == "var" then
    local value, err = random_var(args[1])
    if value == nil then
      return false, err, 2
    end

    return true, value, 0
  end

  return false, "u rand: unknown command '" .. command .. "'\n" .. USAGE, 2
end

return M
```
